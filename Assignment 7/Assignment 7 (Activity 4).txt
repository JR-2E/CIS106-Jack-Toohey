Function Main
    Declare String Choice
    
    Output "Which shape do you need to calculate the area of? A circle, triangle, square, rectangle, or rhombus?"
    Input choice
    If choice = "circle" or choice = "Circle"
        Call ProcessCircle
    False:
        If choice = "rectangle" or choice = "Rectangle" or choice = "square" or choice = "Square" or choice = "rhombus" or choice = "Rhombus"
            Call ProcessRectangle
        False:
            If choice = "triangle" or choice = "Triangle"
                Call ProcessTriangle
            False:
                Output "No valid shape was entered."
            End
        End
    End
End

Function ProcessCircle
    Declare Real CircleArea
    Declare Real Radius
    
    Output "What is the radius of the circle?"
    Input Radius
    Assign CircleArea = Radius * Radius * 3.14159
    Output "The area of the circle is " & CircleArea & " square units."
Return Real CircleArea

Function ProcessRectangle
    Declare Real RectangleArea
    Declare Real RectangleBase
    Declare Real RectangleHeight
    
    Output "What is the base of the shape?"
    Input RectangleBase
    Output "What is the height of the shape?"
    Input RectangleHeight
    Assign RectangleArea = RectangleBase * RectangleHeight
    Output "The area is " & RectangleArea & " square units."
Return Real RectangleArea

Function ProcessTriangle
    Declare Real TriangleArea
    Declare Real TriangleBase
    Declare Real TriangleHeight
    
    Output "Enter triangle base dimension."
    Input TriangleBase
    Output "Enter triangle height dimension."
    Input TriangleHeight
    Assign TriangleArea = TriangleBase * TriangleHeight / 2
    Output "The area of the triangle is " & TriangleArea & " square units."
Return Real TriangleArea
