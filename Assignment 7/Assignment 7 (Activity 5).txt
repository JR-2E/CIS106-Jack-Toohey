Function Main
    Declare Real ShoeSize
    
    Output "Enter your shoe size."
    Input ShoeSize
    If Shoesize <4
        Output "Your sock size is extra small."
    False:
        If ShoeSize >=4 and ShoeSize <=6
            Output "Your sock size is small."
        False:
            If ShoeSize >=7 and ShoeSize <=9
                Output "Your sock size is medium."
            False:
                If ShoeSize >=10 and ShoeSize <=12
                    Output "Your sock size is large."
                False:
                    If ShoeSize >=13
                        Output "Your sock size is extra large."
                    False:
                        Output "No valid shoe size was entered. Shoe size must be a positive number."
                    End
                End
            End
        End
    End
End
